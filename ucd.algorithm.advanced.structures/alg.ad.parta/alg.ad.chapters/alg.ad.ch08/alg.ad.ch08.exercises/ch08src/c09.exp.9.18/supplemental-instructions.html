   public static String cleanup(String text)
   {
      String result = text.replace("&", "&");
      result = result.replace("``", "“"); 
      result = result.replace("''", "“"); 
      result = result.replace("`", "‘"); 
      result = result.replace("'", "’"); 
      result = result.replace("--", "&emdash;"); 
      result = result.replace("<", "<"); 
      result = result.replace(">", ">"); 
      return result;
   }
