public class PurseTester

   {
      Purse a = new Purse();
      a.addCoin("Quarter");
      a.addCoin("Dime");
      a.addCoin("Nickel");
      a.addCoin("Dime");        
      
      Purse b = new Purse();    
      b.addCoin("Nickel");
      b.addCoin("Dime");

      c.addCoin("Quarter");
      c.addCoin("Penny");
      c.addCoin("Nickel");
      c.addCoin("Dime");        
      
      Purse d = new Purse();    
      d.addCoin("Nickel");
      d.addCoin("Dime");   
      d.addCoin("Dime");         
      d.addCoin("Quarter");

      System.out.println(c.sameCoins(d));
      System.out.println("Expected: false");
